function maybeBurst(result) { const now = Date.now(); const isKnownColor = result.name !== "不明"; const colorChanged = result.name !== lastColorName; const enoughTimePassed = now - lastBurstTime > 900; if (isKnownColor && colorChanged && enoughTimePassed) { createBurst(result.particleColor, result.motionType); lastBurstTime = now; } lastColorName = result.name; } function classifyColor(r, g, b) { const hsv = rgbToHsv(r, g, b); const h = hsv.h; const s = hsv.s; const v = hsv.v; if (v < 0.25 || s < 0.18) { return { name: "不明", displayColor: "rgba(0,0,0,0.55)", particleColor: "rgb(255,255,255)", motionType: "none" }; } if (h >= 35 && h <= 70) { return { name: "黄色", displayColor: "rgba(255, 220, 40, 0.85)", particleColor: "rgb(255, 230, 50)", motionType: "sun" }; } if (h >= 15 && h < 35) { return { name: "オレンジ", displayColor: "rgba(255, 140, 40, 0.85)", particleColor: "rgb(255, 145, 45)", motionType: "firework" }; } if (h >= 330 || h < 15) { return { name: "ピンク / 赤", displayColor: "rgba(255, 80, 150, 0.85)", particleColor: "rgb(255, 85, 160)", motionType: "flower" }; } if (h >= 80 && h <= 160) { return { name: "緑", displayColor: "rgba(80, 220, 110, 0.85)", particleColor: "rgb(80, 230, 120)", motionType: "leaf" }; } if (h >= 180 && h <= 250) { return { name: "青", displayColor: "rgba(60, 170, 255, 0.85)", particleColor: "rgb(70, 180, 255)", motionType: "splash" }; } if (h > 250 && h < 330) { return { name: "紫", displayColor: "rgba(180, 100, 255, 0.85)", particleColor: "rgb(190, 110, 255)", motionType: "spiral" }; } return { name: "不明", displayColor: "rgba(0,0,0,0.55)", particleColor: "rgb(255,255,255)", motionType: "none" }; } function createBurst(color, motionType) { const centerX = window.innerWidth / 2; const centerY = window.innerHeight / 2; if (motionType === "sun") { // 黄色：太陽のようにまっすぐ明るく広がる for (let i = 0; i < 120; i++) { const angle = Math.random() * Math.PI * 2; const speed = 4 + Math.random() * 9; particles.push({ x: centerX, y: centerY, vx: Math.cos(angle) * speed, vy: Math.sin(angle) * speed, size: 3 + Math.random() * 8, life: 1, color: color, shape: "circle", gravity: 0.02, drag: 0.985, rotation: 0, spin: 0 }); } } if (motionType === "firework") { // オレンジ：強めに爆発して落ちる for (let i = 0; i < 150; i++) { const angle = Math.random() * Math.PI * 2; const speed = 5 + Math.random() * 13; particles.push({ x: centerX, y: centerY, vx: Math.cos(angle) * speed, vy: Math.sin(angle) * speed, size: 4 + Math.random() * 10, life: 1, color: color, shape: "square", gravity: 0.11, drag: 0.975, rotation: Math.random() * Math.PI, spin: -0.15 + Math.random() * 0.3 }); } } if (motionType === "flower") { // ピンク：花びらのようにふわっと舞う for (let i = 0; i < 90; i++) { const angle = Math.random() * Math.PI * 2; const speed = 2 + Math.random() * 6; particles.push({ x: centerX, y: centerY, vx: Math.cos(angle) * speed, vy: Math.sin(angle) * speed - 1.5, size: 6 + Math.random() * 12, life: 1, color: color, shape: "petal", gravity: 0.035, drag: 0.992, rotation: Math.random() * Math.PI, spin: -0.08 + Math.random() * 0.16 }); } } if (motionType === "leaf") { // 緑：葉っぱのようにゆっくり流れる for (let i = 0; i < 80; i++) { const angle = -Math.PI / 2 + (-0.9 + Math.random() * 1.8); const speed = 1.5 + Math.random() * 5; particles.push({ x: centerX, y: centerY, vx: Math.cos(angle) * speed, vy: Math.sin(angle) * speed, size: 7 + Math.random() * 12, life: 1, color: color, shape: "leaf", gravity: -0.005, drag: 0.99, rotation: Math.random() * Math.PI, spin: -0.06 + Math.random() * 0.12 }); } } if (motionType === "splash") { // 青：水しぶきのように上に跳ねてから落ちる for (let i = 0; i < 100; i++) { const angle = -Math.PI / 2 + (-0.9 + Math.random() * 1.8); const speed = 4 + Math.random() * 10; particles.push({ x: centerX, y: centerY, vx: Math.cos(angle) * speed, vy: Math.sin(angle) * speed, size: 3 + Math.random() * 7, life: 1, color: color, shape: "drop", gravity: 0.16, drag: 0.985, rotation: Math.random() * Math.PI, spin: 0.04 }); } } if (motionType === "spiral") { // 紫：回転しながらゆっくり広がる for (let i = 0; i < 120; i++) { const angle = Math.random() * Math.PI * 2; const speed = 2 + Math.random() * 7; particles.push({ x: centerX, y: centerY, vx: Math.cos(angle) * speed, vy: Math.sin(angle) * speed, size: 4 + Math.random() * 10, life: 1, color: color, shape: "circle", gravity: 0.01, drag: 0.99, rotation: angle, spin: 0.12 }); } } } function updateParticles() { effectCtx.clearRect(0, 0, window.innerWidth, window.innerHeight); const nextParticles = []; for (const particle of particles) { particle.vx *= particle.drag; particle.vy *= particle.drag; particle.x += particle.vx; particle.y += particle.vy; particle.vy += particle.gravity; particle.rotation += particle.spin; particle.life -= 0.016; if (particle.life > 0) { nextParticles.push(particle); effectCtx.save(); effectCtx.globalAlpha = Math.max(particle.life, 0); effectCtx.fillStyle = particle.color; effectCtx.translate(particle.x, particle.y); effectCtx.rotate(particle.rotation); const size = particle.size * particle.life; if (particle.shape === "circle") { effectCtx.beginPath(); effectCtx.arc(0, 0, size, 0, Math.PI * 2); effectCtx.fill(); } if (particle.shape === "square") { effectCtx.fillRect(-size / 2, -size / 2, size, size); } if (particle.shape === "petal") { effectCtx.beginPath(); effectCtx.ellipse(0, 0, size * 0.65, size * 1.25, 0, 0, Math.PI * 2); effectCtx.fill(); } if (particle.shape === "leaf") { effectCtx.beginPath(); effectCtx.ellipse(0, 0, size * 0.55, size * 1.45, 0, 0, Math.PI * 2); effectCtx.fill(); effectCtx.globalAlpha = Math.max(particle.life * 0.45, 0); effectCtx.strokeStyle = "rgba(255,255,255,0.8)"; effectCtx.lineWidth = 1; effectCtx.beginPath(); effectCtx.moveTo(0, -size * 1.1); effectCtx.lineTo(0, size * 1.1); effectCtx.stroke(); } if (particle.shape === "drop") { effectCtx.beginPath(); effectCtx.ellipse(0, 0, size * 0.65, size * 1.2, 0, 0, Math.PI * 2); effectCtx.fill(); } effectCtx.restore(); } } particles = nextParticles; }
